3. Call the API for Buttons and Click Events
   - Explanation Connect to /api/appointments with GET so the sunrise clinic frontend keeps Buttons and Click Events mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load Buttons and Click Events');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Buttons and Click Events records: {records.length}</p>;
   ```

3. Call the API for Charts and Graphs
   - Explanation Connect to /api/appointments with GET so the sunrise clinic frontend keeps Charts and Graphs mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load Charts and Graphs');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Charts and Graphs records: {records.length}</p>;
   ```

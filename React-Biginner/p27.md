3. Call the API for Deleting Records
   - Explanation Connect to /api/appointments/123 with DELETE so the sunrise clinic frontend keeps Deleting Records mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments/123', {
         method: 'DELETE',
       });
       if (!response.ok) throw new Error('Failed to load Deleting Records');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Deleting Records records: {records.length}</p>;
   ```

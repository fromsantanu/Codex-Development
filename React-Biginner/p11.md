3. Call the API for Conditional Rendering
   - Explanation Connect to /api/appointments with GET so the sunrise clinic frontend keeps Conditional Rendering mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load Conditional Rendering');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Conditional Rendering records: {records.length}</p>;
   ```

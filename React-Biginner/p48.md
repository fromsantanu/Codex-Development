3. Call the API for File Upload + Summary UI
   - Explanation Connect to /api/reports with POST so the sunrise clinic frontend keeps File Upload + Summary UI mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const formData = new FormData();
       formData.append('file', new Blob(['summary'], { type: 'text/plain' }));
       const response = await fetch('/api/reports', {
         method: 'POST',
         body: formData,
       });
       if (!response.ok) throw new Error('Failed to load File Upload + Summary UI');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>File Upload + Summary UI records: {records.length}</p>;
   ```

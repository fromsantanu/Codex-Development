3. Call the API for Accessibility Basics
   - Explanation Connect to /api/appointments with GET so the sunrise clinic frontend keeps Accessibility Basics mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load Accessibility Basics');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Accessibility Basics records: {records.length}</p>;
   ```

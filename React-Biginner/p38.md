3. Call the API for Drag and Drop UI
   - Explanation Connect to /api/appointments with GET so the sunrise clinic frontend keeps Drag and Drop UI mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load Drag and Drop UI');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Drag and Drop UI records: {records.length}</p>;
   ```

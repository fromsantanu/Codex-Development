3. Call the API for Understanding Routing

   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load Understanding Routing');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Understanding Routing records: {records.length}</p>;
   ```

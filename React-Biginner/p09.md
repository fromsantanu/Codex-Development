3. Call the API for Inputs and Forms
   - Explanation Connect to /api/appointments with GET so the sunrise clinic frontend keeps Inputs and Forms mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load Inputs and Forms');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Inputs and Forms records: {records.length}</p>;
   ```

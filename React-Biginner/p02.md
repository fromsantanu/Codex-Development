   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load How React Renders UI');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>How React Renders UI records: {records.length}</p>;
   ```

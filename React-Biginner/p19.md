   ```jsx
     const [mode, setMode] = useState('ready');
     return (
       <button onClick={() => setMode(mode === 'ready' ? 'checked' : 'ready')}>
         Storing Values: {mode}
       </button>
     );
   }
   ```
   - Codex help: Ask Codex to add a simple toggle so the Storing Values button flips between ready and checked states.

3. Call the API for Storing Values
   - Explanation Connect to /api/appointments with GET so the sunrise clinic frontend keeps Storing Values mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load Storing Values');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Storing Values records: {records.length}</p>;
   ```

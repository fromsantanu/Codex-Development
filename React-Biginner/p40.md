   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load Maps Integration');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Maps Integration records: {records.length}</p>;
   ```

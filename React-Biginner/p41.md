3. Call the API for Animations
   - Explanation Connect to /api/appointments with GET so the sunrise clinic frontend keeps Animations mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments', {
         method: 'GET',
       });
       if (!response.ok) throw new Error('Failed to load Animations');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Animations records: {records.length}</p>;
   ```

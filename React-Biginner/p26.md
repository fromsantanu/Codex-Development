3. Call the API for Editing and Updating Records
   - Explanation Connect to /api/appointments/123 with PUT so the sunrise clinic frontend keeps Editing and Updating Records mirrored with the backend.
   ```jsx
   const [records, setRecords] = useState([]);
   useEffect(() => {
     async function syncData() {
       const response = await fetch('/api/appointments/123', {
         method: 'PUT',
         headers: { 'Content-Type': 'application/json' },
         body: JSON.stringify({ status: 'confirmed' }),
       });
       if (!response.ok) throw new Error('Failed to load Editing and Updating Records');
       const data = await response.json();
       setRecords(data);
     }
     syncData();
   }, []);
   return <p>Editing and Updating Records records: {records.length}</p>;
   ```
